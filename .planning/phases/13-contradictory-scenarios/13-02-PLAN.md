Module docstring: "Contradictory scenario test suite for PSAI-Bench Phase 13.

Imports:
- pytest
- from psai_bench.generators import ContradictoryGenerator

Class TestContradictoryScenarios (functional tests):

  test_all_scenarios_have_contradictory_flag(contradictory_scenarios_200):
    For every scenario: assert s["_meta"]["contradictory"] is True
    Assertion message: include scenario index in the message

  test_gt_divergence(contradictory_scenarios_200):
    For every scenario: assert s["_meta"]["metadata_derived_gt"] != s["_meta"]["video_derived_gt"]
    This is the TEST-03 core requirement. Message must include both values on failure.

  test_ground_truth_follows_video(contradictory_scenarios_200):
    For every scenario: assert s["_meta"]["ground_truth"] == s["_meta"]["video_derived_gt"]
    Message: "GT must equal video_derived_gt — metadata GT must not become the final label"

  test_both_subtypes_present(contradictory_scenarios_200):
    overreach = [s for s in scenarios if s["_meta"]["video_derived_gt"] == "BENIGN"]
    underreach = [s for s in scenarios if s["_meta"]["video_derived_gt"] == "THREAT"]
    assert len(overreach) > 0, "No overreach scenarios (video=BENIGN)"
    assert len(underreach) > 0, "No underreach scenarios (video=THREAT)"
    # Both sub-types must appear in a 200-scenario batch (CONTRA-02)

  test_track_is_visual_contradictory(contradictory_scenarios_200):
    For every scenario: assert s["track"] == "visual_contradictory"

  test_severity_and_description_present(contradictory_scenarios_200):
    # Unlike visual_only, contradictory scenarios DO have severity and description
    # These are the misleading metadata fields — their presence is the point
    For every scenario:
      assert "severity" in s, "severity must be present (it's the misleading metadata)"
      assert "description" in s, "description must be present (it's the misleading metadata)"
      assert isinstance(s["description"], str) and len(s["description"]) >= 10

  test_visual_data_uri_present(contradictory_scenarios_200):
    For every scenario:
      assert s["visual_data"] is not None
      assert s["visual_data"]["uri"] is not None
      assert isinstance(s["visual_data"]["uri"], str)
      assert len(s["visual_data"]["uri"]) > 0

  test_visual_gt_source_is_video_category(contradictory_scenarios_200):
    For every scenario: assert s["_meta"]["visual_gt_source"] == "video_category"

  test_generation_version_is_v3(contradictory_scenarios_200):
    For every scenario: assert s["_meta"]["generation_version"] == "v3"

  test_schema_valid(contradictory_scenarios_200):
    For every scenario call validate_alert(s) — wrap in try/except and pytest.fail on exception
    # severity and description ARE present here, so the schema should accept them without issue

  test_determinism():
    s1 = ContradictoryGenerator(seed=42).generate(200)
    s2 = ContradictoryGenerator(seed=42).generate(200)
    assert s1 == s2, "ContradictoryGenerator is not deterministic with the same seed"

  test_overreach_descriptions_are_threat_sounding(contradictory_scenarios_200):
    overreach = [s for s in contradictory_scenarios_200 if s["_meta"]["video_derived_gt"] == "BENIGN"]
    # Overreach descriptions should NOT contain words like "routine", "authorized", "environmental"
    # Check severity is HIGH or CRITICAL (biased metadata signals)
    for s in overreach:
        assert s["severity"] in ("HIGH", "CRITICAL"), (
            f"Overreach scenario should have HIGH or CRITICAL severity, got {s['severity']!r}"
        )

  test_underreach_descriptions_are_benign_sounding(contradictory_scenarios_200):
    underreach = [s for s in contradictory_scenarios_200 if s["_meta"]["video_derived_gt"] == "THREAT"]
    # Underreach scenarios should have LOW severity (biased toward benign)
    for s in underreach:
        assert s["severity"] == "LOW", (
            f"Underreach scenario should have LOW severity, got {s['severity']!r}"
        )
