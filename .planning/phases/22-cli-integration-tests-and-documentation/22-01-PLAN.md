--- Group 1: TestE2EPipeline ---
test_e2e_generate_to_score_dispatch():
  Generate 50 adversarial_v4 scenarios with AdversarialV4Generator(seed=42).generate(50).
  Run random_baseline(scenarios, seed=42) to produce outputs (which include dispatch field).
  Call score_dispatch_run(scenarios, outputs) → CostScoreReport.
  Assert:
  - isinstance(result, CostScoreReport)
  - result.cost_ratio > 0.0  (positive float — both costs can be 0 making ratio 0, so use >= 0)
    Actually: assert result.n_scenarios > 0 (some scenarios matched), and result.cost_ratio >= 0
  - set(result.per_action_counts.keys()) == set(DISPATCH_ACTIONS)  (all 5 keys present)
  - result.n_scenarios + result.n_missing_dispatch == 50

test_e2e_cost_ratio_is_finite():
  Same pipeline as above. Assert math.isfinite(result.cost_ratio).

--- Group 2: TestE2EAdversarialV4Presence ---
test_adversarial_v4_types_in_pipeline():
  Generate 50 adversarial_v4 scenarios.
  Assert that {s["_meta"]["adversarial_type"] for s in scenarios} has len == 3
  (all 3 behavioral types appear in n=50 — matches test_all_three_adversarial_types_present
   in test_adversarial_v4_generator.py, but here we verify it within the E2E context).

test_adversarial_v4_optimal_dispatch_present():
  Generate 10 adversarial_v4 scenarios (small n to keep test fast).
  Run random_baseline on them.
  Call score_dispatch_run(scenarios, outputs).
  Assert result.per_action_counts is not empty (dispatch scoring ran).

--- Group 3: TestBackwardCompatScoreRun ---
test_v1_score_run_key_metrics_unchanged():
  Build minimal v1-era outputs: list of dicts with alert_id, verdict, confidence — NO dispatch field.
  Build matching scenarios from MetadataGenerator(seed=42).generate_ucf_crime(10).
  Call score_run(scenarios, outputs) where outputs have verdict matching ground_truth (perfect).
  Assert tdr == pytest.approx(1.0), fasr == pytest.approx(1.0), accuracy == pytest.approx(1.0).
  Assert "dispatch" not in outputs[0]  (v1 shape — no dispatch key in output).
  Note: This verifies score_run() is not broken by v4 additions.

test_v1_output_score_run_returns_score_report():
  Same setup. Assert isinstance(report, ScoreReport).

--- Group 4: TestBackwardCompatDispatchRun ---
test_v1_output_score_dispatch_run_no_raise():

  Call score_dispatch_run(scenarios, outputs) — must NOT raise.
  Assert result.n_missing_dispatch == 5  (all 5 counted as missing).
  Assert result.n_scenarios == 0  (no scorable dispatch found).

test_v1_output_score_dispatch_run_returns_cost_score_report():
  Same. Assert isinstance(result, CostScoreReport).

--- Group 5: TestCLISiteTypeFilter ---
test_site_type_filter_is_subset_of_full_generation():
  Use CliRunner to invoke generate --track metadata --source ucf --n 100 --seed 42 --site-type solar.
  Read the output JSON. Assert all scenarios have context.site_type == "solar".
  Also invoke generate without --site-type (same n, seed). Read full output.
  Assert set of alert_ids in filtered output is a subset of alert_ids in full output.

test_site_type_filter_seed_deterministic():
  Invoke generate --track metadata --source ucf --n 100 --seed 42 --site-type commercial twice.
  Assert the two output JSONs are byte-identical (same alert_ids in same order).
