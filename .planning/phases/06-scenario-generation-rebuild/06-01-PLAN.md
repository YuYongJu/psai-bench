```python
# ---------------------------------------------------------------------------
# V2 Ground Truth Decision Function
# ---------------------------------------------------------------------------
# Implements SCEN-01: GT determined by weighted combination of context signals,
# not by category alone. Implements GT-03: pure function, fully deterministic.
#
# Signal scoring: each signal contributes to a weighted_sum in [-1.25, +1.25].
# Negative = BENIGN evidence, Positive = THREAT evidence.
# GT thresholds: sum > +0.30 → THREAT, sum < -0.30 → BENIGN, else → SUSPICIOUS
#
# SCEN-03 compliance: severity_weight = 0.25 < threshold 0.30.
# Severity alone can never cross the threshold — it always needs support from
# one other signal. This produces ~70% severity-GT correlation, not 100%.

_GT_THRESHOLD = 0.30
_AMBIGUITY_THRESHOLD = 0.10  # |sum| < this → "ambiguous by design"

_ZONE_THREAT_SCORES = {
    "restricted": +0.40,   # restricted zones are inherently high-sensitivity
    "utility": +0.25,
    "perimeter": +0.10,
    "interior": 0.00,
    "parking": -0.15,      # parking lots have expected traffic
}

_TIME_THREAT_SCORES = {
    "night": +0.35,
    "dawn": +0.15,
    "dusk": +0.10,
    "day": -0.20,
}

_SEVERITY_THREAT_SCORES = {
    "CRITICAL": +0.25,
    "HIGH": +0.15,
    "MEDIUM": 0.00,
    "LOW": -0.20,
}

# Device FPR: high FPR = camera cries wolf = evidence against threat
def _device_fpr_score(fpr: float) -> float:
    """Map device FPR to threat score. Linear: FPR=0.05 → +0.13, FPR=0.95 → -0.27."""
    # At fpr=0.05: +0.13. At fpr=0.50: -0.07. At fpr=0.95: -0.27
    return round(0.15 - (fpr * (0.40 / 0.90)), 3)


def assign_ground_truth_v2(
    zone_type: str,
    zone_sensitivity: int,
    time_of_day: str,
    device_fpr: float,
    severity: str,
    badge_access_minutes_ago: int | None,
    rng: "np.random.RandomState",
) -> tuple[str, float, bool]:
    """Assign ground truth using a weighted multi-signal scoring function.

    Args:
        zone_type: Zone type string (perimeter, interior, parking, utility, restricted)
        zone_sensitivity: Zone sensitivity 1-5 (modifies zone score)
        time_of_day: Time of day string (day, night, dawn, dusk)
        device_fpr: Device false positive rate 0.0-1.0
        severity: Alert severity string (LOW, MEDIUM, HIGH, CRITICAL)
        badge_access_minutes_ago: Minutes since last badge grant event, or None if no event
        rng: NumPy RandomState (passed for future stochastic extensions; currently unused)

    Returns:
        tuple: (ground_truth label, weighted_sum, is_ambiguous)
        - ground_truth: "THREAT", "SUSPICIOUS", or "BENIGN"
        - weighted_sum: raw score before threshold (useful for Phase 7 leakage tests)
        - is_ambiguous: True if |weighted_sum| < _AMBIGUITY_THRESHOLD
    """
    # Zone score — base + sensitivity modifier
    # Sensitivity 1-5: multiply base score by factor in [0.6, 1.4]
    zone_base = _ZONE_THREAT_SCORES.get(zone_type, 0.0)
    sensitivity_factor = 0.6 + (zone_sensitivity - 1) * 0.2  # 1→0.6, 3→1.0, 5→1.4
    zone_score = zone_base * sensitivity_factor

    # Time of day score
    time_score = _TIME_THREAT_SCORES.get(time_of_day, 0.0)

    # Device FPR score
    fpr_score = _device_fpr_score(device_fpr)

    # Severity score (SCEN-03: max = 0.25 < threshold 0.30)
    severity_score = _SEVERITY_THREAT_SCORES.get(severity, 0.0)

    # Badge access score — recent badge grant = strong BENIGN signal
    if badge_access_minutes_ago is not None and badge_access_minutes_ago <= 10:
        badge_score = -0.45  # Strong: authorized person just badged in
    elif badge_access_minutes_ago is not None and badge_access_minutes_ago <= 30:
        badge_score = -0.25  # Moderate: access was recent
    else:
        badge_score = 0.0  # No recent badge event = no modifier

    weighted_sum = zone_score + time_score + fpr_score + severity_score + badge_score
    weighted_sum = round(weighted_sum, 4)

    is_ambiguous = abs(weighted_sum) < _AMBIGUITY_THRESHOLD

    if weighted_sum > _GT_THRESHOLD:
        gt = "THREAT"
    elif weighted_sum < -_GT_THRESHOLD:
        gt = "BENIGN"
    else:
        gt = "SUSPICIOUS"

    return gt, weighted_sum, is_ambiguous
```
